       - Create a simple EventEmitter or Map-based subscriber registry:
         ```typescript
         const jobEventSubscribers = new Map<string, Set<(event: JobEvent) => void>>();

         function broadcastJobEvent(userId: string, event: Record<string, unknown>) {
           const subs = jobEventSubscribers.get(userId);
           if (subs) {
             const jobEvent: JobEvent = { event: String(event.event ?? 'update'), data: event };
             for (const cb of subs) cb(jobEvent);
           }
         }

         function onJobEvent(userId: string, callback: (event: JobEvent) => void): () => void {
           if (!jobEventSubscribers.has(userId)) {
             jobEventSubscribers.set(userId, new Set());
           }
           jobEventSubscribers.get(userId)!.add(callback);
           return () => {
             jobEventSubscribers.get(userId)?.delete(callback);
             if (jobEventSubscribers.get(userId)?.size === 0) {
               jobEventSubscribers.delete(userId);
             }
           };
         }
         ```
