    Create `job-event-bus.ts` in the `realtime/` directory (same location as sse-progress.ts):

    1. **Factory function:**
       ```typescript
       import type { JobEvent } from './sse-progress';

       type OnJobEventFn = (userId: string, callback: (event: JobEvent) => void) => () => void;
       type PublishFn = (userId: string, event: Record<string, unknown>) => void;

       function createJobEventBus(): { onJobEvent: OnJobEventFn; publish: PublishFn } {
         const subscribers = new Map<string, Set<(event: JobEvent) => void>>();

         function onJobEvent(userId: string, callback: (event: JobEvent) => void): () => void {
           if (!subscribers.has(userId)) {
             subscribers.set(userId, new Set());
           }
           subscribers.get(userId)!.add(callback);
           return () => {
             subscribers.get(userId)?.delete(callback);
             if (subscribers.get(userId)?.size === 0) {
               subscribers.delete(userId);
             }
           };
         }

         function publish(userId: string, event: Record<string, unknown>): void {
           const subs = subscribers.get(userId);
           if (!subs) return;
           const jobEvent: JobEvent = {
             event: String(event.event ?? 'update'),
             data: event,
           };
           for (const cb of subs) cb(jobEvent);
         }

         return { onJobEvent, publish };
       }
       ```

    2. **Export:** `export { createJobEventBus, type OnJobEventFn, type PublishFn };`

    3. **Spec file:**
       - publish with no subscribers does not throw
       - onJobEvent registers callback, publish delivers event to it
       - unsubscribe removes callback, subsequent publish does not deliver
       - multiple subscribers for same userId all receive events
       - subscribers for different userIds are isolated
       - unsubscribe last subscriber cleans up Map entry (memory hygiene)
