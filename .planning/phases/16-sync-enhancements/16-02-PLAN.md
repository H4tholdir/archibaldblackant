  <files>archibald-web-app/backend/src/db/migrations/007-change-log.sql</files>
  <action>
    Create migration for change tracking tables:

    ```sql
    -- Migration 007: Change log and sync metadata for delta sync
    -- Tracks entity changes for incremental client updates

    CREATE TABLE IF NOT EXISTS shared.change_log (
      id SERIAL PRIMARY KEY,
      entity_type TEXT NOT NULL,
      entity_id TEXT NOT NULL,
      sync_version BIGINT NOT NULL,
      operation TEXT NOT NULL CHECK (operation IN ('insert', 'update', 'delete')),
      is_critical BOOLEAN NOT NULL DEFAULT false,
      created_at BIGINT NOT NULL DEFAULT (EXTRACT(EPOCH FROM NOW()) * 1000)::BIGINT
    );

    CREATE INDEX IF NOT EXISTS idx_change_log_version
      ON shared.change_log(sync_version);
    CREATE INDEX IF NOT EXISTS idx_change_log_entity
      ON shared.change_log(entity_type, entity_id);
    CREATE INDEX IF NOT EXISTS idx_change_log_created
      ON shared.change_log(created_at);

    CREATE TABLE IF NOT EXISTS shared.sync_metadata (
      key TEXT PRIMARY KEY,
      version BIGINT NOT NULL DEFAULT 0,
      last_updated BIGINT NOT NULL DEFAULT (EXTRACT(EPOCH FROM NOW()) * 1000)::BIGINT
    );

    -- Initialize version counters for each entity type
    INSERT INTO shared.sync_metadata (key, version)
    VALUES
      ('products', 0),
      ('prices', 0),
      ('customers', 0),
      ('orders', 0)
    ON CONFLICT (key) DO NOTHING;
    ```

    - change_log records each entity mutation with a monotonically increasing sync_version
    - sync_metadata tracks the current version per entity type
    - is_critical flag for changes that must be pushed immediately (e.g., price changes)
    - entity_type values: 'product', 'price', 'customer', 'order'
  </action>
