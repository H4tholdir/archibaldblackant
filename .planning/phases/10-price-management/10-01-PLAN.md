<feature>
  <name>parseItalianPrice + matchVariant</name>
  <files>archibald-web-app/backend/src/services/price-matching.ts, archibald-web-app/backend/src/services/price-matching.spec.ts</files>
  <behavior>
    parseItalianPrice:
    - "1.234,56 €" → 1234.56
    - "0,75" → 0.75
    - "12.345,00" → 12345.00
    - "100" → 100
    - null → null
    - "" → null
    - "abc" → null
    - "-1.234,56" → null (negative prices invalid)

    matchVariant:
    - Given products=[{id:"X.K2", packageContent:"5 colli"}, {id:"X.K3", packageContent:"1 collo"}], itemSelection="K2" → product with "5 colli"
    - Given products=[{id:"X"}], itemSelection=null → first product (fallback)
    - Given products=[], itemSelection="K2" → null
    - Given products=[{id:"X.K3", packageContent:"1 collo"}], itemSelection="K2" → try suffix match X.K2, if no match → null
  </behavior>
  <implementation>
    parseItalianPrice: Strip non-numeric chars except . , - then convert Italian format (dot=thousand, comma=decimal) to JS number. Reject negative or NaN.

    matchVariant: Map itemSelection to expected packageContent using K2→"5 colli", K3→"1 collo" mapping. Fall back to product ID suffix match. Return null if no match.

    Both are pure functions — no DB, no async, no side effects. Use `import type` for ProductRow if needed for matchVariant signature.
  </implementation>
</feature>
